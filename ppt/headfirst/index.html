config


<script>
app.config.viewport = document.querySelector('.viewport');
app.config.enableNavibar = true;       //导航栏的开关
app.config.enableScroll = true;        //页面滚动的开关
app.config.enableTransition = true;    //转场动画的开关
app.config.templateEngine = {
    compile : function(text) {
    	return Mustache.compile(text);
    },

    render : function(compiled, data) {
    	return compiled(data);
    }
}
</script>

Helloworld Page


        helloworld.tpl
<section>
    <input type="text" id="hello-to"><button id="say-hello">say</button>
</section>
        helloworld.js
app.page.define({
    name: 'helloworld',
    title: 'hello, world',
    template: './templates/helloworld.tpl',
    ready: function() {
        this.fill({});
        var btn = document.querySelector('#say-hello'),
            text = document.querySelector('#hello-to');
        btn.addEventListener('click', function() {
            app.navigation.push('hello/' + encodeURI(text.value));
        });
    }
});

sayhello Page


        sayhello.tpl
<h1><center>Hello, {{name}}</center></h1>
        sayhello.js
app.page.define({
    name: 'sayhello',
    title: 'say hello',
    template: './templates/sayhello.tpl',
    route: 'hello\\/(P<name>[^\\/]+)',
    ready: function(){
        var data = {name:decodeURI(app.navigation.getParameter('name'))};
        this.fill(data);
    }
});
